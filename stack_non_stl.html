Stack


Below is the implementation of Stack as a class in C++


class Stack{
    int size;
    int capacity;
public:
    int *arr;
    Stack(int cap){
        size = 0;
        capacity = cap;
        arr = (int *)malloc(sizeof(int)*capacity);
    }
    void Push(int x){
        if(size == capacity){
            cout << "Stack Overflow" << endl;
        }
        else{
            arr[size] = x;
            size++;
        }
    }
    void Pop(){
        if(IsEmpty()){
            cout << "Stack Underflow" << endl;
        }
        else{
            size--;
        }
    }
    int Top(){
        if(IsEmpty()){
            cout << "Stack Empty" << endl;
            return -1;
        }
        else{
            return arr[size-1];
        }
    }
    bool IsEmpty(){
        if(size == 0){
            return 1;
        }
        return 0;
    }
    void Print(){
        if(IsEmpty()){
            cout << "Stack Empty" << endl;
        }
        else{
            cout << "Stack : ";
            for(int i = 0; i < size; i++){
                cout << arr[i] << " ";
            }
            cout << endl;
        }
    }
};
