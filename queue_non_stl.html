Queue


Below is the implementation of Queue as a class in C++


class Queue{
    int front, rear, size, capacity;
public:
    int *arr;
    Queue(int cap){
        capacity = cap;
        front = 0;
        size = 0;
        rear = capacity - 1;
        arr = (int *)malloc(sizeof(int)*capacity);
    }
    bool IsEmpty(){
        if(size == 0){
            return 1;
        }
        return 0;
    }
    bool IsFull(){
        if(size == capacity){
            return 1;
        }
        return 0;
    }
    void Enqueue(int x){
        if(IsFull()){
            cout << "Queue Overflow" << endl;
        }
        else{
            rear = (rear + 1) % capacity;
            arr[rear] = x;
            size++;
        }
    }
    int Dequeue(){
        if(IsEmpty()){
            cout << "Queue Underflow" << endl;
            return -1;
        }
        else{
            int res = arr[front];
            front = (front + 1) % capacity;
            size--;
            return res;
        }
    }
    int Front(){
        if(IsEmpty()){
            cout << "Queue Empty" << endl;
            return -1;
        }
        else{
            return arr[front];
        }
    }
    int Rear(){
        if(IsEmpty()){
            cout << "Queue Empty" << endl;
            return -1;
        }
        else{
            return arr[rear];
        }
    }
    void Print(){
        if(IsEmpty()){
            cout << "Queue Empty" << endl;
        }
        else{
            cout << "Queue : ";
            if(front <= rear){
                for(int i = front; i <= rear; i++){
                    cout << arr[i] << " ";
                }
            }
            else{
                for(int i = front; i < capacity; i++){
                    cout << arr[i] << " ";
                }
                for(int i = 0; i <= rear; i++){
                    cout << arr[i] << " ";
                }
            }
            cout << endl;
        }
    }
};
