Graph


Below is the implementation of directed Graph as a class using Adjacency List representation in C++


typedef struct AdjListNode{
    int dest;
    AdjListNode *next;
} AdjListNode;

AdjListNode *newAdjNode(int des){
    AdjListNode *newNode = (AdjListNode *)malloc(sizeof(AdjListNode));
    newNode->dest = des;
    newNode->next = NULL;
    return newNode;
}

typedef struct AdjList{
    AdjListNode *head;
} AdjList;

class Graph{
    int V;
    AdjList *arr;
public:
    Graph(int v){
        V = v;
        arr = (AdjList *)malloc(sizeof(AdjList) * V);
        for(int i = 0; i < V; i++){
            arr[i].head = NULL;
        }
    }
    void AddEdge(int src, int dest){
        AdjListNode *newNode = newAdjNode(dest);
        newNode->next = arr[src].head;
        arr[src].head = newNode;
    }
    bool FindIfEdge(int src, int dest){
        AdjListNode *counter = arr[src].head;
        while(counter){
            if(counter->dest == dest){
                return 1;
            }
            else{
                counter = counter->next;
            }
        }
        return 0;
    }
    void PrintGraph(){
        for(int i = 0; i < V; i++){
            AdjListNode *counter = arr[i].head;
            printf("\nAdjacency list of vertex %d\n head ", i);
            while(counter){
                printf("-> %d", counter->dest);
                counter = counter->next;
            }
            printf("\n\n");
        }
    }
};

To represent an undirected graph just add the below code to the AddEdge Function


newNode = newAdjNode(src);
newNode->next = arr[dest].head;
arr[dest].head = newNode;
